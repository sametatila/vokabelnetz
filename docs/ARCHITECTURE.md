### Store Implementation Example

```typescript
// src/app/core/state/auth.store.ts

import { Injectable, computed, signal, effect } from '@angular/core';

export interface AuthState {
  user: User | null;
  tokens: AuthTokens | null;
  loading: boolean;
  error: string | null;
  initialized: boolean;
}

/**
 * AuthStore - Manages authentication state.
 * 
 * SECURITY: Access token stored in memory ONLY (not localStorage).
 * Refresh token managed via HttpOnly cookie by the server.
 * See docs/SECURITY.md for detailed token management strategy.
 */
@Injectable({ providedIn: 'root' })
export class AuthStore {
  
  // Private writable state
  private state = signal<AuthState>({
    user: null,
    accessToken: null,  // Memory only - lost on page refresh (by design)
    loading: false,
    error: null,
    initialized: false
  });

  // Public read-only selectors
  user = computed(() => this.state().user);
  isLoggedIn = computed(() => !!this.state().accessToken);
  loading = computed(() => this.state().loading);
  error = computed(() => this.state().error);
  isInitialized = computed(() => this.state().initialized);

  // Computed derived state
  displayName = computed(() => 
    this.state().user?.displayName || 'Guest'
  );

  constructor(private authService: AuthService) {
    // On app init, try to refresh tokens using HttpOnly cookie
    this.initializeAuth();
  }
  
  /**
   * Initialize auth state on app start.
   * Attempts to get new access token using refresh cookie.
   */
  private async initializeAuth(): Promise<void> {
    this.state.update(s => ({ ...s, loading: true }));
    
    try {
      // Call refresh endpoint - refresh token is in HttpOnly cookie
      const response = await this.authService.refreshTokens().toPromise();
      
      if (response?.accessToken) {
        this.state.update(s => ({
          ...s,
          accessToken: response.accessToken,
          user: response.user,
          initialized: true,
          loading: false
        }));
      } else {
        this.state.update(s => ({ ...s, initialized: true, loading: false }));
      }
    } catch {
      // No valid refresh token - user needs to login
      this.state.update(s => ({ ...s, initialized: true, loading: false }));
    }
  }

  // Actions
  setUser(user: User): void {
    this.state.update(s => ({ ...s, user }));
  }

  setAccessToken(token: string): void {
    // âš ï¸ SECURITY: Token stored in memory only, not localStorage!
    this.state.update(s => ({ ...s, accessToken: token }));
  }

  setLoading(loading: boolean): void {
    this.state.update(s => ({ ...s, loading }));
  }

  logout(): void {
    this.state.set({
      user: null,
      tokens: null,
      loading: false,
      error: null,
      initialized: true
    });
  }
}
```

### Cross-Store Computed State

```typescript
// src/app/core/state/app.state.ts

@Injectable({ providedIn: 'root' })
export class AppState {
  
  constructor(
    private authStore: AuthStore,
    private learningStore: LearningStore,
    private progressStore: ProgressStore
  ) {}

  // Cross-store computed signals
  isReady = computed(() => 
    this.authStore.isInitialized() && 
    this.progressStore.isLoaded()
  );

  canStartSession = computed(() =>
    this.authStore.isLoggedIn() &&
    !this.learningStore.hasActiveSession()
  );

  todayComplete = computed(() =>
    this.progressStore.dailyProgress().wordsLearned >= 
    this.authStore.user()?.dailyGoal
  );
}
```

| Setting | Options | Default | Description |
|---------|---------|---------|-------------|
| `ui_language` | `tr`, `en`, `de` | System language* | Interface/menu language |
| `source_language` | `tr`, `en` | System language* | Translation FROM (what you know) |
| `target_language` | `de` | `de` | Translation TO (always German) |
| `show_both_translations` | `true/false` | `false` | Show TR+EN together |

**\* Default Language Detection Logic:**
```
1. Check browser/system language (navigator.language)
2. If system language is 'tr' or 'tr-TR' â†’ Use Turkish
3. If system language is 'en' or 'en-*' â†’ Use English  
4. Otherwise â†’ Default to English
```

### Language Service

```typescript
// src/app/core/services/language.service.ts

export type UiLanguage = 'tr' | 'en' | 'de';
export type SourceLanguage = 'tr' | 'en';

@Injectable({ providedIn: 'root' })
export class LanguageService {
  
  currentUiLanguage = signal<UiLanguage>(this.detectDefaultLanguage());
  currentSourceLanguage = signal<SourceLanguage>(this.detectDefaultSourceLanguage());
  showBothTranslations = signal<boolean>(false);

  constructor(private translate: TranslateService) {
    // Initialize with detected language
    this.translate.use(this.currentUiLanguage());
  }

  /**
   * Detect default UI language from browser/system settings.
   * Returns 'tr' if Turkish, 'en' otherwise (fallback).
   */
  private detectDefaultLanguage(): UiLanguage {
    const browserLang = navigator.language?.toLowerCase() || '';
    
    if (browserLang.startsWith('tr')) {
      return 'tr';
    }
    
    // Default to English for all other languages
    return 'en';
  }

  /**
   * Detect default source language (learning FROM).
   * Same logic as UI language.
   */
  private detectDefaultSourceLanguage(): SourceLanguage {
    const browserLang = navigator.language?.toLowerCase() || '';
    
    if (browserLang.startsWith('tr')) {
      return 'tr';
    }
    
    return 'en';
  }

  setUiLanguage(lang: UiLanguage): void {
    this.currentUiLanguage.set(lang);
    this.translate.use(lang);
    this.persistSettings();
  }

  setSourceLanguage(lang: SourceLanguage): void {
    this.currentSourceLanguage.set(lang);
    this.persistSettings();
  }

  getWordTranslation(word: { turkish: string; english: string }): string {
    return this.currentSourceLanguage() === 'tr' 
      ? word.turkish 
      : word.english;
  }

  private persistSettings(): void {
    localStorage.setItem('language_settings', JSON.stringify({
      ui: this.currentUiLanguage(),
      source: this.currentSourceLanguage(),
      showBoth: this.showBothTranslations()
    }));
  }

  loadPersistedSettings(): void {
    const saved = localStorage.getItem('language_settings');
    if (saved) {
      const settings = JSON.parse(saved);
      this.currentUiLanguage.set(settings.ui);
      this.currentSourceLanguage.set(settings.source);
      this.showBothTranslations.set(settings.showBoth);
      this.translate.use(settings.ui);
    }
  }
}
```

### Timezone Service

```typescript
// src/app/core/services/timezone.service.ts

@Injectable({ providedIn: 'root' })
export class TimezoneService {
  
  private static readonly DEFAULT_TIMEZONE = 'Europe/Istanbul'; // UTC+2/+3
  
  /**
   * Detect user's system timezone.
   * Falls back to Europe/Istanbul if detection fails.
   */
  detectTimezone(): string {
    try {
      const detected = Intl.DateTimeFormat().resolvedOptions().timeZone;
      
      if (detected && this.isValidTimezone(detected)) {
        return detected;
      }
    } catch (e) {
      console.warn('Timezone detection failed:', e);
    }
    
    return TimezoneService.DEFAULT_TIMEZONE;
  }
  
  /**
   * Check if a timezone string is valid.
   */
  private isValidTimezone(tz: string): boolean {
    try {
      Intl.DateTimeFormat(undefined, { timeZone: tz });
      return true;
    } catch {
      return false;
    }
  }
  
  /**
   * Get current date in user's timezone.
   */
  getCurrentDate(timezone?: string): Date {
    const tz = timezone || this.detectTimezone();
    return new Date(new Date().toLocaleString('en-US', { timeZone: tz }));
  }
  
  /**
   * Check if it's a new day in user's timezone.
   */
  isNewDay(lastActiveAt: Date, timezone?: string): boolean {
    const tz = timezone || this.detectTimezone();
    const now = this.getCurrentDate(tz);
    const last = new Date(lastActiveAt.toLocaleString('en-US', { timeZone: tz }));
    
    return now.toDateString() !== last.toDateString();
  }
}
```

```typescript
// src/app/core/services/audio.service.ts

@Injectable({ providedIn: 'root' })
export class AudioService {
  
  // Static files only for A1 to prevent repo bloat
  private readonly AUDIO_BASE_URL = '/assets/audio/a1';
  private speechSynthesis: SpeechSynthesis | null = null;
  private germanVoice: SpeechSynthesisVoice | null = null;

  status = signal<'idle' | 'loading' | 'playing' | 'error'>('idle');

  constructor() {
    this.initWebSpeechAPI();
  }

  /**
   * Play pronunciation for a word.
   * A1 words: Try static file first, then Web Speech API
   * A2/B1 words: Web Speech API directly (no static files)
   */
  async playWord(word: string, cefrLevel: 'A1' | 'A2' | 'B1'): Promise<void> {
    this.status.set('loading');
    
    // Only A1 has static files (to prevent repo bloat)
    if (cefrLevel === 'A1') {
      const staticUrl = `${this.AUDIO_BASE_URL}/${word.toLowerCase()}.mp3`;
      
      try {
        await this.playAudioFile(staticUrl);
        return;
      } catch {
        // Fall through to Web Speech API
      }
    }
    
    // A2/B1 or A1 fallback: Use Web Speech API
    if (this.germanVoice) {
      this.speakWithTTS(word);
      return;
    }
    
    // No audio available
    this.status.set('error');
  }

  private initWebSpeechAPI(): void {
    if ('speechSynthesis' in window) {
      this.speechSynthesis = window.speechSynthesis;
      
      // Load voices (may be async)
      const loadVoices = () => {
        const voices = this.speechSynthesis!.getVoices();
        this.germanVoice = voices.find(v => v.lang.startsWith('de')) || null;
      };
      
      loadVoices();
      this.speechSynthesis.onvoiceschanged = loadVoices;
    }
  }

  private async playAudioFile(url: string): Promise<void> {
    return new Promise((resolve, reject) => {
      const audio = new Audio(url);
      audio.onended = () => {
        this.status.set('idle');
        resolve();
      };
      audio.onerror = () => reject(new Error('Audio file not found'));
      audio.onplay = () => this.status.set('playing');
      audio.play().catch(reject);
    });
  }

  private speakWithTTS(text: string): void {
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.voice = this.germanVoice;
    utterance.lang = 'de-DE';
    utterance.rate = 0.9;
    
    utterance.onstart = () => this.status.set('playing');
    utterance.onend = () => this.status.set('idle');
    utterance.onerror = () => this.status.set('error');
    
    this.speechSynthesis?.speak(utterance);
  }
}
```

## Design Decisions

### Why Spring Boot 4.0?

| Consideration | Decision |
|--------------|----------|
| Virtual Threads | Java 25 virtual threads for better concurrency |
| Native Compilation | GraalVM support for faster startup (future) |
| Spring Security 7 | Modern security defaults |
| Jakarta EE | Full Jakarta namespace migration |

### Why Angular 21?

| Consideration | Decision |
|--------------|----------|
| Signals | Native reactive primitives |
| Standalone Components | Simplified module structure |
| Control Flow | `@if`, `@for` syntax |
| SSR Ready | Server-side rendering support |

### Why PostgreSQL 18?

| Consideration | Decision |
|--------------|----------|
| JSONB | Flexible schema for word metadata |
| Full-Text Search | Turkish/German language support |
| Extensions | pg_trgm for fuzzy matching |
| Reliability | Mature, well-tested database |
